# Assorted scheme modules

ADD_LIBRARY (exec ExecSCM.cc)

TARGET_LINK_LIBRARIES(exec execution smob)

ADD_GUILE_EXTENSION(SCM_CONFIG exec "opencog-ext-path-exec")

ADD_GUILE_MODULE (FILES
   opencog/exec.scm
   DEPENDS exec

)
